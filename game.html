<body>

    <div class="header">

        <div id="level-info">第 1 關 (難度: 1)</div>

        <div style="font-size: 12px; color: #888;">提示：手動拖曳內容來查看隱藏區域</div>

        <button onclick="changeZoom(-0.2)">－ 縮小</button>

        <button onclick="changeZoom(0.2)">＋ 放大</button>

        <button class="btn-next" onclick="resetLevel()">重置本關</button>

    </div>

    <script>

        let currentLevel = 1;

        let zoomLevel = 1.0;

        let gameState = [];

        const directions = [

            { icon: '↑', dx: 0, dy: -1 },

            { icon: '↓', dx: 0, dy: 1 },

            { icon: '←', dx: -1, dy: 0 },

            { icon: '→', dx: 1, dy: 0 }

        ];

        function initGame() {

            const board = document.getElementById('game-board');

            const info = document.getElementById('level-info');

           

            // 難度邏輯

            let difficulty = (currentLevel % 10 === 0) ? 5 : Math.min(4, 1 + Math.floor((currentLevel % 10) / 2.5));

           

            // 格子數邏輯：隨等級提升，最高到 50x50 (200x200 手機瀏覽器會崩潰)

            let boardSize = Math.min(50, 3 + Math.floor(currentLevel / 2));

           

            info.innerText = `第 ${currentLevel} 關 (難度: ${difficulty}, 尺寸: ${boardSize}x${boardSize})`;

           

            board.innerHTML = '';

            board.style.gridTemplateColumns = `repeat(${boardSize}, 1fr)`;

           

            const blockSize = 40; // 固定每個方塊大小，靠縮放來控制視野

            board.style.width = `${boardSize * (blockSize + 2)}px`;

            // 根據難度決定顏色

            const colors = ['#3498db', '#2ecc71', '#f1c40f', '#e67e22', '#e74c3c'];

            const levelColor = colors[difficulty - 1];

            gameState = [];

            for (let y = 0; y < boardSize; y++) {

                gameState[y] = [];

                for (let x = 0; x < boardSize; x++) {

                    const dir = directions[Math.floor(Math.random() * 4)];

                    const block = document.createElement('div');

                    block.className = 'block';

                    block.style.width = `${blockSize}px`;

                    block.style.height = `${blockSize}px`;

                    block.style.backgroundColor = levelColor;

                    block.innerText = dir.icon;

                    block.onclick = (e) => handleTap(x, y);

                   

                    board.appendChild(block);

                    gameState[y][x] = { dir, element: block };

                }

            }

            updateBoardTransform();

        }

        function handleTap(x, y) {

            const block = gameState[y][x];

            if (!block) return;

            const { dx, dy } = block.dir;

            let cx = x + dx, cy = y + dy;

            let blocked = false;

            while (cx >= 0 && cx < gameState[0].length && cy >= 0 && cy < gameState.length) {

                if (gameState[cy][cx]) { blocked = true; break; }

                cx += dx; cy += dy;

            }

            if (blocked) {

                block.element.classList.add('shake');

                setTimeout(() => block.element.classList.remove('shake'), 200);

            } else {

                block.element.style.transform = `translate(${dx * 800}px, ${dy * 800}px)`;

                block.element.style.opacity = '0';

                gameState[y][x] = null;

                setTimeout(() => {

                    block.element.remove();

                    checkWin();

                }, 400);

            }

        }

        function changeZoom(delta) {

            zoomLevel = Math.max(0.5, Math.min(3, zoomLevel + delta));

            updateBoardTransform();

        }

        function updateBoardTransform() {

            const board = document.getElementById('game-board');

            board.style.transform = `scale(${zoomLevel})`;

        }

        function checkWin() {

            const remains = gameState.flat().some(cell => cell !== null);

            if (!remains) {

                alert("太強了！進入下一關");

                currentLevel++;

                initGame();

            }

        }

        function resetLevel() { initGame(); }

        initGame();
